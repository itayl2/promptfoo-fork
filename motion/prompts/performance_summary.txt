For context, this is the zod schema to conform to:

```typescript
import { z } from "zod";

export const CreativePerformanceSummarySchema = z.object({
  slug: z.string().describe("Unique identifier slug for the ad creative"),
  performanceHighlights: z
    .object({
      text: z
        .string()
        .describe("Short summary of the creative's performance highlights"),
    })
    .describe("Overview of key performance points"),
  metricsAnalysis: z
    .object({
      relevantToClientGoal: z
        .string()
        .describe("How the metrics align with or differ from client goals"),
      outliers: z
        .string()
        .optional()
        .describe("Notable anomalies or outliers in performance metrics"),
      timeComparison: z
        .string()
        .optional()
        .describe(
          "Insights on how metrics changed over different time periods"
        ),
    })
    .describe("Core analysis of performance metrics"),
  spendTrendAnalysis: z
    .object({
      trendText: z
        .string()
        .describe("Narrative about the current spend trend and direction"),
      isFatigued: z
        .boolean()
        .describe(
          "Indicates if the creative is likely experiencing fatigue or not"
        ),
    })
    .describe("Spend analysis and fatigue check"),
  overallSummary: z
    .object({
      winnerLoserStatus: z
        .string()
        .describe(
          "Indicates if the creative is a winner or loser compared to others"
        ),
      relativePerformance: z
        .string()
        .describe("Comparison with other creatives in the account"),
      notableOutliers: z
        .array(z.string())
        .optional()
        .describe("List of metrics or observations that deviate from the norm"),
      timeMetricChanges: z
        .string()
        .optional()
        .describe("How the creative's metrics evolved over time"),
      fatigueAssessment: z
        .string()
        .describe(
          "Additional commentary on whether the creative shows signs of fatigue"
        ),
    })
    .describe(
      "Final summary of the creative's performance and any concluding insights"
    ),
});
```
